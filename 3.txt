//largest odd number
class Solution {
    public String largestOddNumber(String num) {
        String ans = new String();

        for(int i=num.length()-1 ; i>=0 ; i--)
        {
            char ch = num.charAt(i);
            int digit = ch-'0';
            if((digit % 2) == 1){
                ans = num.substring(0,i+1);
                break;
            }
        }
        
        return ans;
    }
}
